create database BidvestBank;

use BidvestBank;

create table BankBranch(
    branch_name varchar(10) not null primary key,
    branch_city varchar(25) not null,
    assets FLOAT not NULL 
);

create table Customer(
    customer_name varchar(25) not null primary key,
    customer_street varchar(25) not null,
    customer_city varchar(25) not null
);

create table CustomerAccount(
    account_number varchar(5) not null primary key,
    branch_name varchar(10) not null,
    balance FLOAT not null
);

create table Loan(
    loan_number varchar(10) not null primary key,
    branch_name varchar(10) not null,
    amount FLOAT not null
);

create table Depositor(
    customer_name varchar(25) not null,
    account_number varchar(10) not null
);

create table Borrower(
    customer_name varchar(25) not null,
    loan_number varchar(10) not null
);

CREATE TABLE Employee (
    employee_name VARCHAR(25) NOT NULL,
    branch_name VARCHAR(10) not null,
    salary FLOAT not null
);
